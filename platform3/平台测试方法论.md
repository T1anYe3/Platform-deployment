### 2.1 服务状态检查

```bash

docker compose ps

# 单个容器健康
docker inspect <container> --format '{{.State.Health.Status}}'

# 等待全部 healthy（Python）
```

```python
import subprocess, time

def wait_healthy(containers, timeout=300):
    for t in range(0, timeout, 5):
        r = subprocess.run(['docker','ps','--format','{{.Names}}\t{{.Status}}'], 
                          capture_output=True, text=True, timeout=10)
        status_map = {}
        for line in r.stdout.strip().split('\n'):
            if '\t' in line:
                name, status = line.split('\t', 1)
                short = name.replace('platform1-', '').replace('platform1-docker-', '')
                status_map[short] = 'healthy' in status
        all_ok = all(status_map.get(c, False) for c in containers)
        if all_ok:
            return True
        time.sleep(5)
    return False
```

### 2.3 TLS / SSL 验证

```python
import socket, ssl

def check_tls(host, port):
    ctx = ssl.create_default_context()
    ctx.check_hostname = False; ctx.verify_mode = ssl.CERT_NONE
    sock = socket.create_connection((host, port), timeout=5)
    try:
        sock = ctx.wrap_socket(sock, server_hostname=host)
        cert = sock.getpeercert()
        issuer = dict(x[0] for x in cert.get('issuer', []))
        return True, issuer.get('commonName', 'unknown')
    except:
        return False, None
    finally:
        sock.close()
```

## 3. ES 数据注入技巧

### 3.1 直接 bulk 注入（绕过 bridge）

当 bridge 未就绪或需要快速验证数据管道时，直接向 ES 注入测试数据：

```python
import json, urllib.request, base64, time, random

ES = 'http://localhost:9200'
# 如果开启了认证
AUTH = base64.b64encode(b'user:password'.encode()).decode()
HEADERS = {'Content-Type': 'application/x-ndjson', 'Authorization': f'Basic {AUTH}'}

docs = []
for i in range(100):
    docs.append({
        'event_id': f'test-{i:04d}',
        'timestamp': int(time.time()) - random.randint(0, 3600),
        'field1': 'value1', 'field2': random.randint(1, 100),
    })

lines = []
for doc in docs:
    ts = time.strftime('%Y.%m.%d', time.gmtime(doc['timestamp']))
    dt = time.strftime('%Y-%m-%dT%H:%M:%SZ', time.gmtime(doc['timestamp']))
    doc['@timestamp'] = dt
    lines.append(json.dumps({'index': {'_index': f'my-index-{ts}', '_id': doc['event_id']}}))
    lines.append(json.dumps(doc))

body = '\n'.join(lines) + '\n'
req = urllib.request.Request(f'{ES}/_bulk', data=body.encode(), headers=HEADERS)
result = json.loads(urllib.request.urlopen(req, timeout=30).read())
print(f'Indexed: {len(result.get("items", []))} docs, errors={result.get("errors")}')
```

## 4. 大规模测试数据生成

### 4.1 GB 级文件生成

```python
import os, json, random, time

def generate_json_file(path, target_mb):
    """生成 JSON Lines 格式测试文件"""
    i = 0
    with open(path, 'w') as f:
        while os.path.getsize(path) / (1024*1024) < target_mb:
            rec = {
                'timestamp': time.strftime('%Y-%m-%dT%H:%M:%SZ'),
                'id': i, 'value': random.uniform(0, 100),
                'category': random.choice(['A', 'B', 'C']),
            }
            f.write(json.dumps(rec) + '\n')
            i += 1
    return i

def generate_binary_file(path, size_mb):
    chunk = os.urandom(1024 * 1024)
    with open(path, 'wb') as f:
        for _ in range(size_mb):
            f.write(chunk)

# 使用：generate_json_file('/tmp/test.json', 256)  # 256 MB
```

### 4.2 攻击流量模拟

```python
import socket, ssl, random

def send_http_request(host, port, method, path, user_agent, body=''):
    """发送单条 HTTP 请求（用于触发 IDS/WAF）"""
    req = f'{method} {path} HTTP/1.1\r\nHost: {host}\r\nUser-Agent: {user_agent}\r\n'
    if body:
        req += f'Content-Length: {len(body)}\r\n'
    req += 'Connection: close\r\n\r\n' + body

    ctx = ssl.create_default_context()
    ctx.check_hostname = False; ctx.verify_mode = ssl.CERT_NONE
    sock = socket.create_connection((host, port), timeout=5)
    if port == 443:
        sock = ctx.wrap_socket(sock, server_hostname=host)
    sock.sendall(req.encode())
    sock.close()

# 攻击向量库
ATTACKS = [
    ('GET', '/?id=1 OR 1=1--', 'sqlmap/1.6'),          # SQL injection
    ('GET', '/../../etc/passwd', 'curl/7.68'),           # Path traversal
    ('POST', '/login', 'Nikto/2.1.6'),                  # Scanner
    ('GET', '/<script>alert(1)</script>', 'Mozilla/5.0'), # XSS
]

for i in range(10000):
    method, path, ua = random.choice(ATTACKS)
    send_http_request('127.0.0.1', 80, method, path, ua)
```

## 5. Bridge / 数据管道测试

### 5.1 Bridge 脚本模式

```python
# 所有 bridge 遵循统一模式：
# 1. ensure_index_template() - 创建 ES 索引映射
# 2. load_state() / save_state() - 增量同步状态
# 3. collect_data() - 从源端拉取
# 4. index_to_es() - 批量写入 ES
# 5. 3 次重试 + 递增退避

import time

MAX_RETRIES = 3
BACKOFF = [5, 15, 30]

def run_with_retry(func):
    for attempt in range(MAX_RETRIES + 1):
        try:
            return func()
        except Exception as e:
            if attempt < MAX_RETRIES:
                wait = BACKOFF[min(attempt, len(BACKOFF)-1)]
                print(f'Retry in {wait}s... ({e})')
                time.sleep(wait)
    raise Exception(f'All {MAX_RETRIES+1} attempts failed')
```

### 5.2 验证 Bridge 是否正确工作

## 6. 安全指标测试

### 6.1 TLS 覆盖率

```python
def benchmark_tls():
    services = [(name, host, port, requires_tls), ...]
    tls_ok = sum(1 for n,h,p,r in services if requires_tls and check_tls(h,p)[0])
    tls_req = sum(1 for _,_,_,r in services if r)
    return round(tls_ok / tls_req * 100, 1)
```

### 6.3 数据吞吐量

```python
import time

def benchmark_throughput(test_file_path):
    size_mb = os.path.getsize(test_file_path) / (1024*1024)
    t0 = time.time()
    with open(test_file_path, 'rb') as f:
        subprocess.run(['docker','exec','-i','container','sh','-c',
            'cat > /data/dest.file'], input=f.read(), capture_output=True, timeout=300)
    elapsed = time.time() - t0
    return round(size_mb / elapsed, 1)  # MB/s
```

## 9. Platform 1 踩过的坑（Platform 2/3 避免）

| # | 问题 | 原因 | 解决 |
|---|------|------|------|
| 1 | Kibana SPA 截图 Loading | Chrome headless `--virtual-time-budget` 太短 | 设为 30000ms + 截图后验证大小 > 20KB |
| 2 | Vault bridge 无法读日志 | bridge 容器未挂载 `vault-data` volume | docker-compose 添加 `vault-data:/vault/data:ro` |
| 3 | NiFi API 认证失败 | NiFi 2.x 用 JWT Token 而非 Basic Auth | 先 POST `/access/token` 获取 token |
| 4 | ES auth 后 bridge 不工作 | bridge 脚本未传 `Authorization` 头 | 所有 ES 请求加上 `Authorization: Basic xxx` |
| 5 | ES bulk 报 `ip` 类型错误 | 空字符串不能存入 `ip` 类型字段 | 空值跳过或改用 `keyword` 类型 |
| 6 | 中文路径导致 ES 启动失败 | ES 不兼容非 ASCII 路径 | 必须用纯 ASCII 路径 |
| 7 | Suricata host 模式不工作 | WSL2/Docker Desktop 网络隔离 | 直接注入 eve.json 测试链路 |
| 8 | `curl -k` 不够 | 部分自签证书需 `--connect-timeout` | Python ssl context + `check_hostname=False` |
